1. 当场景生成很多对象时
2. 将不用的`GameObject`放进缓存池要用时再拿出来
3. 避免`GC`时产生的卡顿

```cs
public class PoolData
{
    GameObject fatherObj;//场景上的父节点
    public List<GameObject> Datalist;//抽屉

    //当外部需要创建这个抽屉时
    //初始化抽屉
    //创建场景上的父节点(与物体同名) 挂载到场景上的pool
    //并放入第一个物体
    public PoolData(GameObject poolObj, GameObject obj)
    {
        Datalist = new List<GameObject>();
        fatherObj = new GameObject(obj.name);
        fatherObj.transform.parent = poolObj.transform;
        PushObj(obj);
    }

    //拿到第一个抽屉里的物体
    //将物体 取消与场景上的父节点 的关联
    //激活物体
    //从抽屉里取出
    public GameObject GetObj()
    {
        GameObject obj = Datalist[0];
        obj.transform.parent = null;
        obj.SetActive(true);
        Datalist.RemoveAt(0);
        return obj;
    }

    //将物体失活
    //设置父级为fatherObj
    //放进抽屉里
    public void PushObj(GameObject obj)
    {
        obj.SetActive(false);
        obj.transform.parent = fatherObj.transform;
        Datalist.Add(obj);
    }
}


/// <summary>
/// 缓存池
/// </summary>
public class PoolManager : BaseManager<PoolManager>
{
    public GameObject poolObj;//场景上代表物体的空节点

    //定义一个缓存池
    //用Dictionary 和 list
    public Dictionary<string, PoolData> poolDic = new Dictionary<string, PoolData>();

    //从池里拿东西 通过 抽屉名字
    public GameObject GetObj(string name)
    {
        GameObject obj = null;
        //有这个抽屉 且抽屉里的东西大于0
        //将抽屉里第一个物品给出去 并 移出pool
        if (poolDic.ContainsKey(name) && poolDic[name].Datalist.Count > 0)
        {
            obj = poolDic[name].GetObj();
        }
        //没有就创建这个物体
        //给出去 但是 不能保存到pool
        else
        {
            obj = GameObject.Instantiate(Resources.Load<GameObject>(name));
        }

        //为obj的name赋值这样就可以之后通过name拿到obj
        //给出去的物体激活
        obj.name = name;
        obj.SetActive(true);
        return obj;
    }

    //将物品放到场景上的父节点里
    //将物品放到池里
    public void PushObj(string name, GameObject obj)
    {
        if (poolObj == null)
            poolObj = new GameObject("Pool");

        //使这个物品失活
        obj.SetActive(false);

        //存放物品
        //有这个存放物品的抽屉
        if (poolDic.ContainsKey(name))
        {
            poolDic[name].PushObj(obj);
        }
        //没有这个抽屉
        //创建这个抽屉并存放物品
        else
        {
            poolDic.Add(name, new PoolData(poolObj, obj));
        }

    }

    //场景转换时清理缓存池
    public void Clear()
    {
        poolDic.Clear();
        poolObj = null;
    }

}
```
